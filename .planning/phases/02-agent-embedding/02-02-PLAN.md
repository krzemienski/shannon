### Task 2 — Write `build/embed-skills.py` (the build script)

**Spec:**
```python
#!/usr/bin/env python3
"""build/embed-skills.py — Shannon v1 agent embedding build step.

Reads each agent's manifest.yml; for each embedded_skills entry:
  - copies v1/shannon/skills/<skill>/ → v1/shannon/agents/<agent>/_built/skills/<skill>/
  - preserves cascade references/ directories

Idempotent: re-running overwrites _built/ to match current manifest.
"""

import os, shutil, yaml
from pathlib import Path

REPO_ROOT = Path(__file__).resolve().parent.parent
AGENTS_DIR = REPO_ROOT / "agents"
SKILLS_DIR = REPO_ROOT / "skills"

def main():
    agent_dirs = [d for d in AGENTS_DIR.iterdir() if d.is_dir()]
    if not agent_dirs:
        print("No agents found")
        return 1

    errors = []
    summary = []
    for agent_dir in sorted(agent_dirs):
        manifest_path = agent_dir / "manifest.yml"
        if not manifest_path.is_file():
            errors.append(f"{agent_dir.name}: missing manifest.yml")
            continue

        manifest = yaml.safe_load(manifest_path.read_text())
        embedded = manifest.get("embedded_skills", [])
        if not embedded:
            errors.append(f"{agent_dir.name}: empty embedded_skills")
            continue

        built_dir = agent_dir / "_built" / "skills"
        if built_dir.exists():
            shutil.rmtree(built_dir, ignore_errors=True)
        built_dir.mkdir(parents=True, exist_ok=True)

        copied = 0
        for skill_name in embedded:
            src = SKILLS_DIR / skill_name
            dst = built_dir / skill_name
            if not src.is_dir():
                errors.append(f"{agent_dir.name}: skill `{skill_name}` not found in skills/")
                continue
            shutil.copytree(src, dst, ignore=shutil.ignore_patterns(".DS_Store"))
            copied += 1
        summary.append(f"{agent_dir.name}: embedded {copied}/{len(embedded)}")

    print("\n".join(summary))
    if errors:
        print("ERRORS:")
        for e in errors: print(f"  {e}")
        return 1
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
```
